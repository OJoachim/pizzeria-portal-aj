import React from 'react';
const TableRow = ({ booking }) => {

  function renderButtons(table) {
    if(table.booking) {
      return (
        <Button className={styles.tabstatus} component={Link} to={`${process.env.PUBLIC_URL}/tables/booking/${table.booking}`}>
          <Typography className={styles.spanone}>see</Typography> 
          {table.booking}
        </Button>);
    }
    else if(table.event) {
      return (
        <Button className={styles.tabstatus} component={Link} to={`${process.env.PUBLIC_URL}/tables/event/${table.eventId}`}>
          <Typography className={styles.spanone}>see</Typography> 
          {table.booking}
        </Button>);
    } else {
      return (
        <>
          <Button className={styles.tabstatus} component={Link} to={`${process.env.PUBLIC_URL}/tables/booking/new`}>
            <Typography className={styles.spanone}>add booking</Typography>
          </Button>
          <Button className={styles.tabstatus} component={Link} to={`${process.env.PUBLIC_URL}/tables/event/new`}>
            <Typography className={styles.spanone}>add event</Typography>
          </Button>
        </>
      );
    }

  }

  return (
    <TableRow>
      <TableCell component="th" scope="row">
        {booking.time}
      </TableCell>
      {booking.tables.map(table => (
        <TableCell>{renderButtons(table)}</TableCell>
      ))}
    </TableRow>);
};
